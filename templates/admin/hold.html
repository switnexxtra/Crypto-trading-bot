Active

`; break; case "pending": statusBadge = `

Pending

`; break; case "verified": statusBadge = `

Verified

`; break; default: statusBadge = `

${account.status || "None"}

`; break; } let row = ` ${index + 1} ${account.user_id} ${account.username} ${account.email} ${statusBadge}
`; tableBody.innerHTML += row; }); }) .catch(error => console.error("Error fetching users:", error)); } // Confirm delete button in delete modal document.getElementById('confirmDelete').addEventListener('click', function () { const transactionId = this.getAttribute('data-transaction-id'); // Send AJAX request to delete transaction fetch('/admin/delete_transaction', { method: 'POST', headers: { 'Content-Type': 'application/json', 'X-CSRFToken': getCsrfToken() // Function to get CSRF token from cookie }, body: JSON.stringify({ transaction_id: transactionId }) }) .then(response => response.json()) .then(data => { if (data.success) { // Close the modal const modal = bootstrap.Modal.getInstance(document.getElementById('deleteModal')); modal.hide(); // Refresh the table searchUsers(); // Reuse the search function to refresh with current search term } else { alert('Error deleting transaction: ' + data.message); } }) .catch(error => { console.error('Error:', error); }); });
